25 std::unique_ptr<NewtonSolver> NewtonSolver::getSolver(realtype *t, AmiVector *x,
26  Solver &simulationSolver,
27  Model *model) {
28 
29  std::unique_ptr<NewtonSolver> solver;
30 
31  switch (simulationSolver.getLinearSolver()) {
32 
33  /* DIRECT SOLVERS */
34  case LinearSolver::dense:
35  solver.reset(new NewtonSolverDense(t, x, model));
36  break;
37 
38  case LinearSolver::band:
39  throw NewtonFailure(AMICI_NOT_IMPLEMENTED, "getSolver");
40 
41  case LinearSolver::LAPACKDense:
42  throw NewtonFailure(AMICI_NOT_IMPLEMENTED, "getSolver");
43 
44  case LinearSolver::LAPACKBand:
45  throw NewtonFailure(AMICI_NOT_IMPLEMENTED, "getSolver");
46 
47  case LinearSolver::diag:
48  throw NewtonFailure(AMICI_NOT_IMPLEMENTED, "getSolver");
49 
50  /* ITERATIVE SOLVERS */
51  case LinearSolver::SPGMR:
52  throw NewtonFailure(AMICI_NOT_IMPLEMENTED, "getSolver");
53 
54  case LinearSolver::SPBCG:
55  solver.reset(new NewtonSolverIterative(t, x, model));
56  break;
57 
58  case LinearSolver::SPTFQMR:
59  throw NewtonFailure(AMICI_NOT_IMPLEMENTED, "getSolver");
60 
61  /* SPARSE SOLVERS */
62  case LinearSolver::SuperLUMT:
63  throw NewtonFailure(AMICI_NOT_IMPLEMENTED, "getSolver");
64  case LinearSolver::KLU:
65  solver.reset(new NewtonSolverSparse(t, x, model));
66  break;
67  default:
68  throw NewtonFailure(AMICI_NOT_IMPLEMENTED, "getSolver");
69  }
70  solver->atol_ = simulationSolver.getAbsoluteTolerance();
71  solver->rtol_ = simulationSolver.getRelativeTolerance();
72  solver->max_lin_steps_ = simulationSolver.getNewtonMaxLinearSteps();
73  solver->max_steps = simulationSolver.getNewtonMaxSteps();
74  solver->damping_factor_mode_ = simulationSolver.getNewtonDampingFactorMode();
75  solver->damping_factor_lower_bound =
76  simulationSolver.getNewtonDampingFactorLowerBound();
77  if (simulationSolver.getLinearSolver() == LinearSolver::SPBCG)
78  solver->num_lin_steps_.resize(simulationSolver.getNewtonMaxSteps(), 0);
79 
80  return solver;
81 }

94 void NewtonSolver::computeNewtonSensis(AmiVectorArray &sx) {
95  prepareLinearSystem(0, -1);
96  model_->fdxdotdp(*t_, *x_, dx_);
97 
98  if (model_->pythonGenerated) {
99  for (int ip = 0; ip < model_->nplist(); ip++) {
100  N_VConst(0.0, sx.getNVector(ip));
101 
102  // copy explicit version
103  if (model_->ndxdotdp_explicit > 0) {
104  auto col = model_->dxdotdp_explicit.indexptrs();
105  auto row = model_->dxdotdp_explicit.indexvals();
106  auto data_ptr = model_->dxdotdp_explicit.data();
107  for (sunindextype iCol = col[model_->plist(ip)];
108  iCol < col[model_->plist(ip) + 1]; ++iCol)
109  sx.at(static_cast<int>(row[iCol]), ip) -= data_ptr[iCol];
110  }
111 
112  // copy implicit version
113  if (model_->ndxdotdp_implicit > 0) {
114  auto col = model_->dxdotdp_implicit.indexptrs();
115  auto row = model_->dxdotdp_implicit.indexvals();
116  auto data_ptr = model_->dxdotdp_implicit.data();
117  for (sunindextype iCol = col[model_->plist(ip)];
118  iCol < col[model_->plist(ip) + 1]; ++iCol)
119  sx.at(static_cast<int>(row[iCol]), ip) -= data_ptr[iCol];
120  }
121 
122  solveLinearSystem(sx[ip]);
123  }
124  } else {
125  for (int ip = 0; ip < model_->nplist(); ip++) {
126  for (int ix = 0; ix < model_->nx_solver; ix++)
127  sx.at(ix,ip) = -model_->dxdotdp.at(ix, ip);
128 
129  solveLinearSystem(sx[ip]);
130  }
131  }
132 }

254 void NewtonSolverIterative::prepareLinearSystem(int ntry, int nnewt) {
255  newton_try_ = ntry;
256  i_newton_ = nnewt;
257  if (nnewt == -1) {
258  throw NewtonFailure(AMICI_NOT_IMPLEMENTED,
259  "Linear solver SPBCG does not support sensitivity "
260  "computation for steady state problems.");
261  }
262 
263  // Get the Jacobian and its diagonal for preconditioning
264  model_->fJ(*t_, 0.0, *x_, dx_, xdot_, ns_J_.get());
265  model_->fJDiag(*t_, ns_Jdiag_, 0.0, *x_, dx_);
266 
267  // Ensure positivity of entries in ns_Jdiag
268  ns_p_.set(1.0);
269  N_VAbs(ns_Jdiag_.getNVector(), ns_Jdiag_.getNVector());
270  N_VCompare(1e-15, ns_Jdiag_.getNVector(), ns_tmp_.getNVector());
271  N_VLinearSum(-1.0, ns_tmp_.getNVector(), 1.0, ns_p_.getNVector(), ns_tmp_.getNVector());
272  N_VLinearSum(1.0, ns_Jdiag_.getNVector(), 1.0, ns_tmp_.getNVector(), ns_Jdiag_.getNVector());
273 }

277 void NewtonSolverIterative::prepareLinearSystemB(int ntry, int nnewt) {
278  newton_try_ = ntry;
279  i_newton_ = nnewt;
280  if (nnewt == -1) {
281  throw AmiException("Linear solver SPBCG does not support sensitivity "
282  "computation for steady state problems.");
283  }
284 
285  // Get the Jacobian and its diagonal for preconditioning
286  model_->fJB(*t_, 0.0, *x_, dx_, xB_, dxB_, xdot_, ns_J_.get());
287 
288  // Get the diagonal and ensure negativity of entries is ns_J. Note that diag(JB) = -diag(J).
289  model_->fJDiag(*t_, ns_Jdiag_, 0.0, *x_, dx_);
290 
291  ns_p_.set(1.0);
292  N_VAbs(ns_Jdiag_.getNVector(), ns_Jdiag_.getNVector());
293  N_VCompare(1e-15, ns_Jdiag_.getNVector(), ns_tmp_.getNVector());
294  N_VLinearSum(-1.0, ns_tmp_.getNVector(), 1.0, ns_p_.getNVector(), ns_tmp_.getNVector());
295  N_VLinearSum(1.0, ns_Jdiag_.getNVector(), 1.0, ns_tmp_.getNVector(), ns_Jdiag_.getNVector());
296 
297  std::transform(ns_Jdiag_.data(), ns_Jdiag_.data()+ns_Jdiag_.getLength(),
298  ns_Jdiag_.data(), std::negate<realtype>());
299 }

310 void NewtonSolverIterative::linsolveSPBCG(int ntry, int nnewt,
311  AmiVector &ns_delta) {
312  xdot_ = ns_delta;
313  xdot_.minus();
314 
315  // Initialize for linear solve
316  ns_p_.reset();
317  ns_v_.reset();
318  ns_delta.reset();
319  ns_tmp_.reset();
320  double rho = 1.0;
321  double omega = 1.0;
322  double alpha = 1.0;
323 
324  ns_J_.multiply(ns_Jv_.getNVector(), ns_delta.getNVector());
325 
326  // ns_r = xdot - ns_Jv;
327  N_VLinearSum(-1.0, ns_Jv_.getNVector(), 1.0, xdot_.getNVector(), ns_r_.getNVector());
328  N_VDiv(ns_r_.getNVector(), ns_Jdiag_.getNVector(), ns_r_.getNVector());
329  ns_rt_ = ns_r_;
330 
331  for (int i_linstep = 0; i_linstep < max_lin_steps_;
332  i_linstep++) {
333  // Compute factors
334  double rho1 = rho;
335  rho = N_VDotProd(ns_rt_.getNVector(), ns_r_.getNVector());
336  double beta = rho * alpha / (rho1 * omega);
337 
338  // ns_p = ns_r + beta * (ns_p - omega * ns_v);
339  N_VLinearSum(1.0, ns_p_.getNVector(), -omega, ns_v_.getNVector(), ns_p_.getNVector());
340  N_VLinearSum(1.0, ns_r_.getNVector(), beta, ns_p_.getNVector(), ns_p_.getNVector());
341 
342  // ns_v = J * ns_p
343  ns_v_.reset();
344  ns_J_.multiply(ns_v_.getNVector(), ns_p_.getNVector());
345  N_VDiv(ns_v_.getNVector(), ns_Jdiag_.getNVector(), ns_v_.getNVector());
346 
347  // Compute factor
348  alpha = rho / N_VDotProd(ns_rt_.getNVector(), ns_v_.getNVector());
349 
350  // ns_h = ns_delta + alpha * ns_p;
351  N_VLinearSum(1.0, ns_delta.getNVector(), alpha, ns_p_.getNVector(),
352  ns_h_.getNVector());
353  // ns_s = ns_r - alpha * ns_v;
354  N_VLinearSum(1.0, ns_r_.getNVector(), -alpha, ns_v_.getNVector(), ns_s_.getNVector());
355 
356  // ns_t = J * ns_s
357  ns_t_.reset();
358  ns_J_.multiply(ns_t_.getNVector(), ns_s_.getNVector());
359  N_VDiv(ns_t_.getNVector(), ns_Jdiag_.getNVector(), ns_t_.getNVector());
360 
361  // Compute factor
362  omega = N_VDotProd(ns_t_.getNVector(), ns_s_.getNVector()) / N_VDotProd(ns_t_.getNVector(), ns_t_.getNVector());
363 
364  // ns_delta = ns_h + omega * ns_s;
365  N_VLinearSum(1.0, ns_h_.getNVector(), omega, ns_s_.getNVector(),
366  ns_delta.getNVector());
367  // ns_r = ns_s - omega * ns_t;
368  N_VLinearSum(1.0, ns_s_.getNVector(), -omega, ns_t_.getNVector(), ns_r_.getNVector());
369 
370  // Compute the (unscaled) residual
371  N_VProd(ns_r_.getNVector(), ns_Jdiag_.getNVector(), ns_r_.getNVector());
372  double res = sqrt(N_VDotProd(ns_r_.getNVector(), ns_r_.getNVector()));
373 
374  // Test convergence
375  if (res < atol_) {
376  // Write number of steps needed
377  num_lin_steps_.at(nnewt) = i_linstep + 1;
378 
379  // Return success
380  return;
381  }
382 
383  // Scale back
384  N_VDiv(ns_r_.getNVector(), ns_Jdiag_.getNVector(), ns_r_.getNVector());
385  }
386  throw NewtonFailure(AMICI_CONV_FAILURE, "linsolveSPBCG");
387 }
