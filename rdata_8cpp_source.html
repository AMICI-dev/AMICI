234 void ReturnData::processForwardProblem(ForwardProblem const &fwd, Model &model,
235  ExpData const *edata) {
236  if (edata)
237  initializeObjectiveFunction();
238 
239  auto initialState = fwd.getInitialSimulationState();
240  if (initialState.x.getLength() == 0)
241  return; // if x wasn't set forward problem failed during initialization
242  readSimulationState(initialState, model);
243 
244  if (!x0.empty()) {
245  model.fx_rdata(x_rdata_, x_solver_);
246  writeSlice(x_rdata_, x0);
247  }
248 
249  if (!sx0.empty()) {
250  model.fsx_rdata(sx_rdata_, sx_solver_);
251  for (int ip = 0; ip < nplist; ip++)
252  writeSlice(sx_rdata_[ip], slice(sx0, ip, nx));
253  }
254 
255  // process timpoint data
256  realtype tf = fwd.getFinalTime();
257  for (int it = 0; it < model.nt(); it++) {
258  if (model.getTimepoint(it) <= tf) {
259  readSimulationState(fwd.getSimulationStateTimepoint(it), model);
260  getDataOutput(it, model, edata);
261  } else {
262  // check for integration failure but consider postequilibration
263  if (!std::isinf(model.getTimepoint(it)))
264  invalidate(it);
265  }
266  }
267 
268  // process event data
269  if (nz > 0) {
270  auto rootidx = fwd.getRootIndexes();
271  for (int iroot = 0; iroot <= fwd.getEventCounter(); iroot++) {
272  readSimulationState(fwd.getSimulationStateEvent(iroot), model);
273  getEventOutput(iroot, t_, rootidx.at(iroot), model, edata);
274  }
275  }

402 void ReturnData::processBackwardProblem(ForwardProblem const &fwd,
403  BackwardProblem const &bwd,
404  SteadystateProblem const *preeq,
405  Model &model) {
406  if (sllh.empty())
407  return;
408  readSimulationState(fwd.getInitialSimulationState(), model);
409 
410  std::vector<realtype> llhS0(model.nJ * model.nplist(), 0.0);
411  auto xB = bwd.getAdjointState();
412  auto xQB = bwd.getAdjointQuadrature();
413 
414  if (preeq && preeq->hasQuadrature()) {
415  handleSx0Backward(model, *preeq, xQB);
416  } else {
417  handleSx0Forward(model, llhS0, xB);
418  }
419 
420  for (int iJ = 0; iJ < model.nJ; iJ++) {
421  for (int ip = 0; ip < model.nplist(); ip++) {
422  if (iJ == 0) {
423  sllh.at(ip) -= llhS0[ip] + xQB[ip * model.nJ];
424  } else {
425  s2llh.at(iJ - 1 + ip * (model.nJ - 1)) -=
426  llhS0[ip + iJ * model.nplist()] + xQB[iJ + ip * model.nJ];
427  }
428  }
429  }

432 void ReturnData::handleSx0Backward(const Model &model,
433  SteadystateProblem const &preeq,
434  AmiVector &xQB) const {
435  /* If preequilibration is run in adjoint mode, the scalar product of sx0
436  with its adjoint counterpart (see handleSx0Forward()) is not necessary:
437  the actual simulation is "extended" by the preequilibration time.
438  At initialization (at t=-inf), the adjoint state is in steady state (= 0)
439  and so is the scalar product. Instead of the scalar product, the
440  quadratures xQB from preequilibration contribute to the gradient
441  (see example notebook on equilibration for further documentation). */
442  auto xQBpreeq = preeq.getAdjointQuadrature();
443  for (int ip = 0; ip < model.nplist(); ++ip)
444  xQB[ip] += xQBpreeq[ip];
445 }

447 void ReturnData::handleSx0Forward(const Model &model,
448  std::vector<realtype> &llhS0,
449  AmiVector &xB) const {
450  /* If preequilibration is run in forward mode or is not needed, then adjoint
451  sensitivity analysis still needs the state sensitivities at t=0 (sx0),
452  to compute the gradient. For each parameter, the scalar product of sx0
453  with its adjoint counterpart contributes to the gradient. */
454  for (int iJ = 0; iJ < model.nJ; iJ++) {
455  if (iJ == 0) {
456  for (int ip = 0; ip < model.nplist(); ++ip) {
457  llhS0[ip] = 0.0;
458  for (int ix = 0; ix < model.nxtrue_solver; ++ix) {
459  llhS0[ip] += xB[ix] * sx_solver_.at(ix, ip);
460  }
461  }
462  } else {
463  for (int ip = 0; ip < model.nplist(); ++ip) {
464  llhS0[ip + iJ * model.nplist()] = 0.0;
465  for (int ix = 0; ix < model.nxtrue_solver; ++ix) {
466  llhS0[ip + iJ * model.nplist()] +=
467  xB[ix + iJ * model.nxtrue_solver] * sx_solver_.at(ix, ip) +
468  xB[ix] * sx_solver_.at(ix + iJ * model.nxtrue_solver, ip);
469  }
470  }
471  }
472  }

542 void ReturnData::applyChainRuleFactorToSimulationResults(const Model &model) {
543  // chain-rule factor: multiplier for am_p
544  std::vector<realtype> coefficient(nplist, 1.0);
545  std::vector<realtype> pcoefficient(nplist, 1.0);
546 
547  std::vector<realtype> unscaledParameters = model.getParameters();
548  unscaleParameters(unscaledParameters, model.getParameterScale(),
549  unscaledParameters);
550 
551  std::vector<realtype> augcoefficient(np, 1.0);
552 
553  if (sensi == SensitivityOrder::second && o2mode == SecondOrderMode::full) {
554  for (int ip = 0; ip < np; ++ip) {
555  switch (pscale[ip]) {
556  case ParameterScaling::log10:
557  augcoefficient.at(ip) = unscaledParameters.at(ip) * log(10);
558  break;
559  case ParameterScaling::ln:
560  augcoefficient.at(ip) = unscaledParameters.at(ip);
561  break;
562  case ParameterScaling::none:
563  break;
564  }
565  }
566  }
567 
568  for (int ip = 0; ip < nplist; ++ip) {
569  switch (pscale[model.plist(ip)]) {
570  case ParameterScaling::log10:
571  coefficient.at(ip) = log(10.0);
572  pcoefficient.at(ip) =
573  unscaledParameters.at(model.plist(ip)) * log(10);
574  break;
575  case ParameterScaling::ln:
576  pcoefficient.at(ip) = unscaledParameters.at(model.plist(ip));
577  break;
578  case ParameterScaling::none:
579  break;
580  }
581  }
