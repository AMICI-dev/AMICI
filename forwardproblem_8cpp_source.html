46 void ForwardProblem::workForwardProblem() {
47  FinalStateStorer fss(this);
48  if(model->nx_solver == 0){
49  return;
50  }
51 
52  auto presimulate = edata && edata->t_presim > 0;
53 
54  /* if preequilibration was done, model was already initialized */
55  if (!preequilibrated_)
56  model->initialize(x_, dx_, sx_, sdx_,
57  solver->getSensitivityOrder() >=
58  SensitivityOrder::first);
59 
60  /* compute initial time and setup solver for (pre-)simulation */
61  auto t0 = model->t0();
62  if (presimulate)
63  t0 -= edata->t_presim;
64  solver->setup(t0, model, x_, dx_, sx_, sdx_);
65 
66  /* perform presimulation if necessary */
67  if (presimulate) {
68  if (solver->computingASA())
69  throw AmiException("Presimulation with adjoint sensitivities"
70  " is currently not implemented.");
71  handlePresimulation();
72  t_ = model->t0();
73  }
74  /* when computing adjoint sensitivity analysis with presimulation,
75  we need to store sx after the reinitialization after preequilibration
76  but before reinitialization after presimulation. As presimulation with ASA
77  will not update sx, we can simply extract the values here.*/
78  if (solver->computingASA() && presimulate)
79  sx_ = solver->getStateSensitivity(model->t0());
80 
81  if (presimulate || preequilibrated_)
82  solver->updateAndReinitStatesAndSensitivities(model);
83 
84  // update x0 after computing consistence IC/reinitialization
85  x_ = solver->getState(model->t0());
86  /* when computing forward sensitivities, we generally want to update sx
87  after presimulation/preequilibration, and if we didn't do either this also
88  wont harm. when computing ASA, we only want to update here, if we didn't
89  update before presimulation (if applicable).
90  */
91  if (solver->computingFSA() || (solver->computingASA() && !presimulate ))
92  sx_ = solver->getStateSensitivity(model->t0());
93 
94 
95  /* store initial state and sensitivity*/
96  initial_state_ = getSimulationState();
97 
98  /* loop over timepoints */
99  for (it_ = 0; it_ < model->nt(); it_++) {
100  auto nextTimepoint = model->getTimepoint(it_);
101 
102  if (std::isinf(nextTimepoint))
103  break;
104 
105  if (nextTimepoint > model->t0()) {
106  // Solve for nextTimepoint
107  while (t_ < nextTimepoint) {
108  int status = solver->run(nextTimepoint);
109  solver->writeSolution(&t_, x_, dx_, sx_, dx_);
110  /* sx will be copied from solver on demand if sensitivities
111  are computed */
112  if (status == AMICI_ILL_INPUT) {
113  /* clustering of roots => turn off rootfinding */
114  solver->turnOffRootFinding();
115  } else if (status == AMICI_ROOT_RETURN) {
116  handleEvent(&tlastroot_, false);
117  }
118  }
119  }
120  handleDataPoint(it_);
121  }
122 
123  /* fill events */
124  if (model->nz > 0 && model->nt() > 0) {
125  fillEvents(model->nMaxEvent());
126  }
127 }

140 void ForwardProblem::handleEvent(realtype *tlastroot, const bool seflag) {
141  /* store heaviside information at event occurence */
142  model->froot(t_, x_, dx_, rootvals_);
143 
144  /* store timepoint at which the event occured*/
145  discs_.push_back(t_);
146 
147  /* extract and store which events occured */
148  if (!seflag) {
149  solver->getRootInfo(roots_found_.data());
150  }
151  root_idx_.push_back(roots_found_);
152 
153  rval_tmp_ = rootvals_;
154 
155  if (!seflag) {
156  /* only check this in the first event fired, otherwise this will always
157  * be true */
158  if (t_ == *tlastroot) {
159  throw AmiException("AMICI is stuck in an event, as the initial"
160  "step-size after the event is too small. To fix "
161  "this, increase absolute and relative tolerances!");
162  }
163  *tlastroot = t_;
164  }
165 
166  if(model->nz>0)
167  storeEvent();
168 
169  /* if we need to do forward sensitivities later on we need to store the old
170  * x and the old xdot */
171  if (solver->getSensitivityOrder() >= SensitivityOrder::first) {
172  /* store x and xdot to compute jump in sensitivities */
173  x_old_ = x_;
174  }
175  if (solver->computingFSA()) {
176  model->fxdot(t_, x_, dx_, xdot_);
177  xdot_old_ = xdot_;
178  dx_old_ = dx_;
179  /* compute event-time derivative only for primary events, we get
180  * into trouble with multiple simultaneously firing events here (but
181  * is this really well defined then?), in that case just use the
182  * last ie and hope for the best. */
183  if (!seflag) {
184  for (int ie = 0; ie < model->ne; ie++) {
185  if (roots_found_.at(ie) == 1) {
186  /* only consider transitions false -> true */
187  model->getEventTimeSensitivity(stau_, t_, ie, x_, sx_);
188  }
189  }
190  }
191  } else if (solver->computingASA()) {
192  /* store x to compute jump in discontinuity */
193  x_disc_.push_back(x_);
194  xdot_disc_.push_back(xdot_);
195  xdot_old_disc_.push_back(xdot_old_);
196  }
197 
198  model->updateHeaviside(roots_found_);
199 
200  applyEventBolus();
201 
202  if (solver->computingFSA()) {
203  /* compute the new xdot */
204  model->fxdot(t_, x_, dx_, xdot_);
205  applyEventSensiBolusFSA();
206  }
207 
208  int secondevent = 0;
209 
210  /* check whether we need to fire a secondary event */
211  model->froot(t_, x_, dx_, rootvals_);
212  for (int ie = 0; ie < model->ne; ie++) {
213  /* the same event should not trigger itself */
214  if (roots_found_.at(ie) == 0) {
215  /* check whether there was a zero-crossing */
216  if (0 > rval_tmp_.at(ie) * rootvals_.at(ie)) {
217  if (rval_tmp_.at(ie) < rootvals_.at(ie)) {
218  roots_found_.at(ie) = 1;
219  } else {
220  roots_found_.at(ie) = -1;
221  }
222  secondevent++;
223  } else {
224  roots_found_.at(ie) = 0;
225  }
226  } else {
227  /* don't fire the same event again */
228  roots_found_.at(ie) = 0;
229  }
230  }
231  /* fire the secondary event */
232  if (secondevent > 0) {
233  handleEvent(tlastroot, true);
234  }
235 
236  /* only reinitialise in the first event fired */
237  if (!seflag) {
238  solver->reInit(t_, x_, dx_);
239  if (solver->computingFSA()) {
240  solver->sensReInit(sx_, sdx_);
241  }
242  }
243 }

245 void ForwardProblem::storeEvent() {
246  if (t_ == model->getTimepoint(model->nt() - 1)) {
247  // call from fillEvent at last timepoint
248  model->froot(t_, x_, dx_, rootvals_);
249  for (int ie = 0; ie < model->ne; ie++) {
250  roots_found_.at(ie) = (nroots_.at(ie) < model->nMaxEvent()) ? 1 : 0;
251  }
252  root_idx_.push_back(roots_found_);
253  }
254 
255  if (getRootCounter() < getEventCounter()) {
256  /* update stored state (sensi) */
257  event_states_.at(getRootCounter()) = getSimulationState();
258  } else {
259  /* add stored state (sensi) */
260  event_states_.push_back(getSimulationState());
261  }
262 
263  /* EVENT OUTPUT */
264  for (int ie = 0; ie < model->ne; ie++) {
265  /* only look for roots of the rootfunction not discontinuities */
266  if (nroots_.at(ie) >= model->nMaxEvent())
267  continue;
268 
269  /* only consider transitions false -> true or event filling */
270  if (roots_found_.at(ie) != 1 &&
271  t_ != model->getTimepoint(model->nt() - 1)) {
272  continue;
273  }
274 
275  if (edata && solver->computingASA())
276  model->getAdjointStateEventUpdate(slice(dJzdx_, nroots_.at(ie),
277  model->nx_solver * model->nJ),
278  ie, nroots_.at(ie), t_, x_, *edata);
279 
280  nroots_.at(ie)++;
281  }
282 
283  if (t_ == model->getTimepoint(model->nt() - 1)) {
284  // call from fillEvent at last timepoint
285  // loop until all events are filled
286  fillEvents(model->nMaxEvent());
287  }
288 }
