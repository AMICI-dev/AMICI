249 function [ubw,lbw] = ami_bandwidth(M)
250 /* ami_bandwidth implements a bandwidth function for older versionsn of MATLAB
251  *
252  * Parameters:
253  * M: matrix for which the bandwidth is to be computed
254  *
255  * Return values:
256  * ubw: upper matrix bandwidth
257  * lbw: lower matrix bandwidth */
258 if(isempty(M) || isempty(find(M)))
259  ubw = 0;
260  lbw = 0;
261 else
262  [i,j] = find(M);
263  ubw = max(max(j-i),0);
