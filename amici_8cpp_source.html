98 AmiciApplication::runAmiciSimulation(Solver& solver,
99  const ExpData* edata,
100  Model& model,
101  bool rethrow)
102 {
103  /* Applies condition-specific model settings and restores them when going
104  * out of scope */
105  ConditionContext cc1(&model, edata, FixedParameterContext::simulation);
106 
107  std::unique_ptr<ReturnData> rdata = std::make_unique<ReturnData>(solver,
108  model);
109 
110  if (model.nx_solver <= 0) {
111  return rdata;
112  }
113 
114  std::unique_ptr<SteadystateProblem> preeq {};
115  std::unique_ptr<ForwardProblem> fwd {};
116  std::unique_ptr<BackwardProblem> bwd {};
117  std::unique_ptr<SteadystateProblem> posteq {};
118 
119  try {
120  if (solver.getPreequilibration() ||
121  (edata && !edata->fixedParametersPreequilibration.empty())) {
122  ConditionContext cc2(
123  &model, edata, FixedParameterContext::preequilibration
124  );
125 
126  preeq = std::make_unique<SteadystateProblem>(solver, model);
127  preeq->workSteadyStateProblem(&solver, &model, -1);
128  }
129 
130 
131  fwd = std::make_unique<ForwardProblem>(edata, &model, &solver,
132  preeq.get());
133  fwd->workForwardProblem();
134 
135 
136  if (fwd->getCurrentTimeIteration() < model.nt()) {
137  posteq = std::make_unique<SteadystateProblem>(solver, model);
138  posteq->workSteadyStateProblem(&solver, &model,
139  fwd->getCurrentTimeIteration());
140  }
141 
142 
143  if (edata && solver.computingASA()) {
144  fwd->getAdjointUpdates(model, *edata);
145  if (posteq) {
146  posteq->getAdjointUpdates(model, *edata);
147  posteq->workSteadyStateBackwardProblem(&solver, &model,
148  bwd.get());
149  }
150 
151 
152  bwd = std::make_unique<BackwardProblem>(*fwd, posteq.get());
153  bwd->workBackwardProblem();
154 
155 
156  if (preeq) {
157  ConditionContext cc2(&model, edata,
158  FixedParameterContext::preequilibration);
159  preeq->workSteadyStateBackwardProblem(&solver, &model,
160  bwd.get());
161  }
162  }
163 
164  rdata->status = AMICI_SUCCESS;
165 
166  } catch (amici::IntegrationFailure const& ex) {
167  rdata->status = ex.error_code;
168  if (rethrow)
169  throw;
170  warningF("AMICI:simulation",
171  "AMICI forward simulation failed at t = %f:\n%s\n",
172  ex.time,
173  ex.what());
174  } catch (amici::IntegrationFailureB const& ex) {
175  rdata->status = ex.error_code;
176  if (rethrow)
177  throw;
178  warningF(
179  "AMICI:simulation",
180  "AMICI backward simulation failed when trying to solve until t = %f"
181  " (see message above):\n%s\n",
182  ex.time,
183  ex.what());
184  } catch (amici::AmiException const& ex) {
185  rdata->status = AMICI_ERROR;
186  if (rethrow)
187  throw;
188  warningF("AMICI:simulation",
189  "AMICI simulation failed:\n%s\nError occured in:\n%s",
190  ex.what(),
191  ex.getBacktrace());
192  }
193 
194  rdata->processSimulationObjects(preeq.get(), fwd.get(), bwd.get(),
195  posteq.get(), model, solver, edata);
196  return rdata;
197 }

199 std::vector<std::unique_ptr<ReturnData>>
200 AmiciApplication::runAmiciSimulations(const Solver& solver,
201  const std::vector<ExpData*>& edatas,
202  const Model& model,
203  bool failfast,
204 #if defined(_OPENMP)
205  int num_threads
206 #else
207  int /* num_threads */
208 #endif
209 
210 )
211 {
212  std::vector<std::unique_ptr<ReturnData>> results(edatas.size());
213  // is set to true if one simulation fails and we should skip the rest.
214  // shared across threads.
215  bool skipThrough = false;
216 
217 #if defined(_OPENMP)
218 #pragma omp parallel for num_threads(num_threads)
219 #endif
220  for (int i = 0; i < (int)edatas.size(); ++i) {
221  auto mySolver = std::unique_ptr<Solver>(solver.clone());
222  auto myModel = std::unique_ptr<Model>(model.clone());
223 
224  /* if we fail we need to write empty return datas for the python
225  interface */
226  if (skipThrough) {
227  ConditionContext conditionContext(myModel.get(), edatas[i]);
228  results[i] =
229  std::unique_ptr<ReturnData>(new ReturnData(solver, model));
230  } else {
231  results[i] = runAmiciSimulation(*mySolver, edatas[i], *myModel);
232  }
233 
234  skipThrough |= failfast && results[i]->status < 0;
235  }
236 
237  return results;
238 }
