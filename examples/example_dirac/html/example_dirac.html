    sig = 1e-2;
    delta_num = @(tau) exp(-1/2*(tau/sig).^2)/(sqrt(2*pi)*sig);

    ode_system = @(t,x,p,k) [-p(1)*x(1)+delta_num(t-p(2));

FINITE DIFFERENCES

    eps = 1e-4;
    xi = log10(p);
    for ip = 1:4;
        xip = xi;
        xip(ip) = xip(ip) + eps;
        solp = simulate_model_dirac(t,xip,k,[],options);
        sx_fd(:,:,ip) = (solp.x - sol.x)/eps;
        sy_fd(:,:,ip) = (solp.y - sol.y)/eps;
    end
