408 function result = isnewer(ver1str,ver2str)
409  /* isnewer checks whether the version indicated in ver1str is newer than
410  * the on in ver2str
411  *
412  * Parameters:
413  * ver1str: version string 1, this should be a string in the format %d.%d.%d @type string
414  * ver2str: version string 1, this should be a string in the format %d.%d.%d @type string
415  *
416  * Return values:
417  * result: flag indicating whether ver1 ist newer than ver2 @type boolean */
418 
419  ver1Parts = getParts(ver1str);
420  ver2Parts = getParts(ver2str);
421  if ver2Parts(1) ~= ver1Parts(1) /* major version */
422 
423  result = ver2Parts(1) < ver1Parts(1);
424  elseif ver2Parts(2) ~= ver1Parts(2) /* minor version */
425 
426  result = ver2Parts(2) < ver1Parts(2);
427  else /* revision version */
428 
429  result = ver2Parts(3) < ver1Parts(3);
430  end

432 function parts = getParts(V)
433  /* getParts takes an input version string and returns an array
434  * containing the major minor and subminor version number
435  *
436  * Parameters:
437  * V: version string, this should be a string in the format %d.%d.%d @type string
438  *
439  * Return values:
440  * parts: array containing the version numbers @type double */
441 
442  parts = sscanf(V, " %d.%d.%d ")^t;
443  if length(parts) < 3
444  parts(3) = 0; /* zero-fills to 3 elements */
445 
446  end
