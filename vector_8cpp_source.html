124 void AmiVectorArray::flatten_to_vector(std::vector<realtype> &vec) const {
125  int n_outer = static_cast<int>(vec_array_.size());
126  if (n_outer == 0)
127  return; // nothing to do ...
128  int n_inner = vec_array_.at(0).getLength();
129 
130  if (static_cast<int>(vec.size()) != n_inner * n_outer) {
131  throw AmiException("Dimension of AmiVectorArray (%ix%i) does not "
132  "match target vector dimension (%u)",
133  n_inner, n_outer, vec.size());
134  }
135 
136  for (int outer = 0; outer < n_outer; ++outer) {
137  for (int inner = 0; inner < n_inner; ++inner)
138  vec.at(inner + outer * n_inner) = this->at(inner, outer);
139  }
140 }
