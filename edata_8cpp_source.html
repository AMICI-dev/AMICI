61 ExpData::ExpData(ReturnData const& rdata, std::vector<realtype> sigma_y,
62  std::vector<realtype> sigma_z)
63  : ExpData(rdata.nytrue, rdata.nztrue, rdata.nmaxevent, rdata.ts)
64 {
65  if (sigma_y.size() != (unsigned) nytrue_ && sigma_y.size() != (unsigned) nytrue_*nt())
66  throw AmiException("Dimension of sigma_y must be %d or %d, was %d", nytrue_, nytrue_*nt(), sigma_y.size());
67 
68  if (sigma_z.size() != (unsigned) nztrue_ && sigma_z.size() != (unsigned) nztrue_*nmaxevent_)
69  throw AmiException("Dimension of sigma_z must be %d or %d, was %d", nztrue_, nztrue_*nmaxevent_, sigma_z.size());
70 
71  std::random_device rd{};
72  std::mt19937 gen{rd()};
73 
74  realtype sigma;
75 
76  checkSigmaPositivity(sigma_y, "sigma_y");
77  checkSigmaPositivity(sigma_z, "sigma_z");
78 
79  for (int iy = 0; iy < nytrue_; ++iy) {
80  for (int it = 0; it < nt(); ++it) {
81  sigma = sigma_y.size() == (unsigned) nytrue_ ? sigma_y.at(iy) : sigma_y.at(iy + nytrue_ * it);
82  std::normal_distribution<> e{0, sigma};
83  observed_data_.at(iy + nytrue_ * it) = rdata.y.at(iy + rdata.ny * it) + e(gen);
84  observed_data_std_dev_.at(iy + nytrue_ * it) = sigma;
85  }
86  }
87 
88  for (int iz = 0; iz < nztrue_; ++iz) {
89  for (int ie = 0; ie < nmaxevent_; ++ie) {
90  sigma = sigma_z.size() == (unsigned) nztrue_ ? sigma_z.at(iz) : sigma_z.at(iz + nztrue_ * ie);
91  std::normal_distribution<> e{0, sigma};
92  observed_events_.at(iz + rdata.nztrue * ie) = rdata.z.at(iz + rdata.nz * ie) + e(gen);
93  observed_data_std_dev_.at(iz + rdata.nztrue * ie) = sigma;
94  }
95  }

351 void ConditionContext::applyCondition(const ExpData *edata,
352  FixedParameterContext fpc)
353 {
354  if(!edata)
355  return;
356 
357  // this needs to go first, otherwise nplist will not have the right
358  // dimension for all other fields that depend on Model::nplist
359  if(!edata->plist.empty())
360  model_->setParameterList(edata->plist);
361 
362  // this needs to go second as setParameterScale will reset sx0
363  if(!edata->pscale.empty()) {
364  if(edata->pscale.size() != (unsigned) model_->np())
365  throw AmiException("Number of parameters (%d) in model does not"
366  " match ExpData (%zd).",
367  model_->np(), edata->pscale.size());
368  model_->setParameterScale(edata->pscale);
369 
370  }
371 
372  if(!edata->x0.empty()) {
373  if(edata->x0.size() != (unsigned) model_->nx_rdata)
374  throw AmiException("Number of initial conditions (%d) in model does"
375  " not match ExpData (%zd).",
376  model_->nx_rdata, edata->x0.size());
377  model_->setInitialStates(edata->x0);
378  }
379 
380  if(!edata->sx0.empty()) {
381  if(edata->sx0.size() != (unsigned) model_->nx_rdata * model_->nplist())
382  throw AmiException("Number of initial conditions sensitivities (%d)"
383  " in model does not match ExpData (%zd).",
384  model_->nx_rdata * model_->nplist(),
385  edata->sx0.size());
386  model_->setInitialStateSensitivities(edata->sx0);
387  }
388 
389  if(!edata->parameters.empty()) {
390  if(edata->parameters.size() != (unsigned) model_->np())
391  throw AmiException("Number of parameters (%d) in model does not"
392  " match ExpData (%zd).",
393  model_->np(), edata->parameters.size());
394  model_->setParameters(edata->parameters);
395  }
396 
397  switch (fpc) {
398  case FixedParameterContext::simulation:
399  if (!edata->fixedParameters.empty()) {
400  // fixed parameter in model are superseded by those provided in
401  // edata
402  if (edata->fixedParameters.size()
403  != (unsigned)model_->nk())
404  throw AmiException("Number of fixed parameters (%d) in model does"
405  "not match ExpData (%zd).",
406  model_->nk(), edata->fixedParameters.size());
407  model_->setFixedParameters(edata->fixedParameters);
408  }
409  break;
410  case FixedParameterContext::preequilibration:
411  if (!edata->fixedParametersPreequilibration.empty()) {
412  // fixed parameter in model are superseded by those provided in
413  // edata
414  if (edata->fixedParametersPreequilibration.size() !=
415  (unsigned)model_->nk())
416  throw AmiException("Number of fixed parameters (%d) in model does"
417  "not match ExpData (preequilibration) (%zd).",
418  model_->nk(),
419  edata->fixedParametersPreequilibration.size());
420  model_->setFixedParameters(edata->fixedParametersPreequilibration);
421  }
422  break;
423  case FixedParameterContext::presimulation:
424  if (!edata->fixedParametersPresimulation.empty()) {
425  // fixed parameter in model are superseded by those provided in
426  // edata
427  if (edata->fixedParametersPresimulation.size()
428  != (unsigned)model_->nk())
429  throw AmiException("Number of fixed parameters (%d) in model does"
430  " not match ExpData (presimulation) (%zd).",
431  model_->nk(),
432  edata->fixedParametersPresimulation.size());
433  model_->setFixedParameters(edata->fixedParametersPresimulation);
434  }
435  break;
436  }
437 
438  if(edata->nt()) {
439  // fixed parameter in model are superseded by those provided in edata
440  model_->setTimepoints(edata->getTimepoints());
441  }
442 
443  model_->setReinitializeFixedParameterInitialStates(
444  edata->reinitializeFixedParameterInitialStates);
445 }
