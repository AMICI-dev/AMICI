26  xdot_old_disc_(fwd.getRHSBeforeDiscontinuities()),
27  sx0_(fwd.getStateSensitivity()),

31  dJydx_(fwd.getDJydx()),
32  dJzdx_(fwd.getDJzdx()) {
33  /* complement dJydx from postequilibration. This should overwrite
34  * anything but only fill in previously 0 values, as only non-inf
35  * timepoints are filled from fwd.
36  */
37  for (int it = 0; it < fwd.model->nt(); it++) {
38  if (std::isinf(fwd.model->getTimepoint(it))) {
39  if (!posteq)
40  throw AmiException("Model has non-finite timpoint but, "
41  "postequilibration did not run");
42 
43  /* copy adjoint update to postequilibration */
44  writeSlice(slice(posteq->getDJydx(), it,
45  fwd.model->nx_solver * fwd.model->nJ),
46  slice(dJydx_, it,
47  fwd.model->nx_solver * fwd.model->nJ));
48 
49  /* If adjoint sensis were computed, copy also quadratures */
50  xQB_.reset();
51  xQB_ = posteq->getEquilibrationQuadratures();
52  }

58 void BackwardProblem::workBackwardProblem() {
59 
60  if (model_->nx_solver <= 0 ||
61  solver_->getSensitivityOrder() < SensitivityOrder::first ||
62  solver_->getSensitivityMethod() != SensitivityMethod::adjoint ||
63  model_->nplist() == 0) {
64  return;
65  }
66 
67  int it = model_->nt() - 1;
68  /* If we have posteq, infty timepoints were already treated */
69  for (int jt = model_->nt() - 1; jt >= 0; jt--)
70  if (std::isinf(model_->getTimepoint(jt)))
71  --it;
72 
73  /* initialize state vectors, depending on postequilibration */
74  model_->initializeB(xB_, dxB_, xQB_, it < model_->nt() - 1);
75 
76  if ((it >= 0 || !discs_.empty()) && model_->getTimepoint(it) > model_->t0())
77  {
78  handleDataPointB(it);
79  solver_->setupB(&which, model_->getTimepoint(it), model_, xB_, dxB_, xQB_);
80  --it;
81 
82  while (it >= 0 || discs_.size() > 0) {
83  /* check if next timepoint is a discontinuity or a data-point */
84  double tnext = getTnext(it);
85 
86  if (tnext < t_) {
87  solver_->runB(tnext);
88  solver_->writeSolutionB(&t_, xB_, dxB_, xQB_, which);
89  }
90 
91  /* handle discontinuity */
92  if (tnext > model_->getTimepoint(it)) {
93  handleEventB();
94  }
95 
96  /* handle data-point */
97  if (tnext == model_->getTimepoint(it)) {
98  handleDataPointB(it);
99  it--;
100  }
101 
102  /* reinit states */
103  solver_->reInitB(which, t_, xB_, dxB_);
104  solver_->quadReInitB(which, xQB_);
105  }
106  }
107 
108  /* we still need to integrate from first datapoint to tstart */
109  if (t_ > model_->t0()) {
110  /* solve for backward problems */
111  solver_->runB(model_->t0());
112  solver_->writeSolutionB(&t_, xB_, dxB_, xQB_, which);
113  }
114 
115  if (edata_ && edata_->t_presim > 0) {
116  ConditionContext cc(model_, edata_, FixedParameterContext::presimulation);
117  solver_->runB(model_->t0() - edata_->t_presim);
118  solver_->writeSolutionB(&t_, xB_, dxB_, xQB_, which);
119  }
120 }

123 void BackwardProblem::handleEventB() {
124  auto rootidx = root_idx_.back();
125  this->root_idx_.pop_back();
126 
127  auto x_disc = this->x_disc_.back();
128  this->x_disc_.pop_back();
129 
130  auto xdot_disc = this->xdot_disc_.back();
131  this->xdot_disc_.pop_back();
132 
133  auto xdot_old_disc = this->xdot_old_disc_.back();
134  this->xdot_old_disc_.pop_back();
135 
136  for (int ie = 0; ie < model_->ne; ie++) {
137 
138  if (rootidx[ie] == 0) {
139  continue;
140  }
141 
142  model_->addAdjointQuadratureEventUpdate(xQB_, ie, t_, x_disc, xB_,
143  xdot_disc,
144  xdot_old_disc);
145  model_->addAdjointStateEventUpdate(xB_, ie, t_, x_disc,
146  xdot_disc,
147  xdot_old_disc);
148 
149  if (model_->nz > 0) {
150  for (int ix = 0; ix < model_->nxtrue_solver; ++ix) {
151  for (int iJ = 0; iJ < model_->nJ; ++iJ) {
152  xB_[ix + iJ * model_->nxtrue_solver] +=
153  dJzdx_[iJ + ( ix + nroots_[ie] * model_->nx_solver )
154  * model_->nJ];
155  }
156  }
157  }
158 
159  nroots_[ie]--;
160  }
161 
162  model_->updateHeavisideB(rootidx.data());
163 }

165 void BackwardProblem::handleDataPointB(const int it) {
166  solver_->storeDiagnosisB(which);
167 
168  for (int ix = 0; ix < model_->nxtrue_solver; ix++) {
169  for (int iJ = 0; iJ < model_->nJ; iJ++)
170  // we only need the 1:nxtrue_solver (not the nx_true) slice here!
171  xB_[ix + iJ * model_->nxtrue_solver] +=
172  dJydx_[iJ + ( ix + it * model_->nx_solver ) * model_->nJ];
173  }
