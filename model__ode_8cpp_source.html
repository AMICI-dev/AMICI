370 void Model_ODE::fqBdot(realtype t, N_Vector x, N_Vector xB, N_Vector qBdot) {
371  /* initialize with zeros */
372  N_VConst(0.0, qBdot);
373  fdxdotdp(t, x);
374 
375  if (pythonGenerated) {
376  /* call multiplication */
377  if (ndxdotdp_explicit > 0)
378  dxdotdp_explicit.multiply(qBdot, xB, state_.plist, true);
379  if (ndxdotdp_implicit > 0)
380  dxdotdp_implicit.multiply(qBdot, xB, state_.plist, true);
381  N_VScale(-1.0, qBdot, qBdot);
382  } else {
383  /* was matlab generated */
384  for (int ip = 0; ip < nplist(); ip++) {
385  for (int ix = 0; ix < nxtrue_solver; ix++)
386  NV_Ith_S(qBdot, ip * nJ) -= NV_Ith_S(xB, ix) * dxdotdp.at(ix, ip);
387  // second order part
388  for (int iJ = 1; iJ < nJ; iJ++)
389  for (int ix = 0; ix < nxtrue_solver; ix++)
390  NV_Ith_S(qBdot, ip * nJ + iJ) -=
391  NV_Ith_S(xB, ix) * dxdotdp.at(ix + iJ * nxtrue_solver, ip) +
392  NV_Ith_S(xB, ix + iJ * nxtrue_solver) * dxdotdp.at(ix, ip);
393  }
394  }

437 void Model_ODE::fsxdot(realtype t, N_Vector x, int ip, N_Vector sx,
438  N_Vector sxdot) {
439 
440  /* sxdot is just the total derivative d(xdot)dp,
441  so we just call dxdotdp and copy the stuff over */
442  if (ip == 0) {
443  // we only need to call this for the first parameter index will be
444  // the same for all remaining
445  fdxdotdp(t, x);
446  fJSparse(t, x, J_.get());
447  }
448  if (pythonGenerated) {
449  /* copy dxdotdp and the implicit version over */
450  // initialize
451  N_VConst(0.0, sxdot);
452  realtype *sxdot_tmp = N_VGetArrayPointer(sxdot);
453 
454  // copy explicit version
455  if (ndxdotdp_explicit > 0) {
456  auto col_exp = dxdotdp_explicit.indexptrs();
457  auto row_exp = dxdotdp_explicit.indexvals();
458  auto data_exp_ptr = dxdotdp_explicit.data();
459  for (sunindextype i = col_exp[plist(ip)]; i < col_exp[plist(ip) + 1]; ++i)
460  sxdot_tmp[row_exp[i]] += data_exp_ptr[i];
461  }
462 
463  // copy implicit version
464  if (ndxdotdp_implicit > 0) {
465  auto col_imp = dxdotdp_implicit.indexptrs();
466  auto row_imp = dxdotdp_implicit.indexvals();
467  auto data_imp_ptr = dxdotdp_implicit.data();
468  for (sunindextype i = col_imp[plist(ip)]; i < col_imp[plist(ip) + 1]; ++i)
469  sxdot_tmp[row_imp[i]] += data_imp_ptr[i];
470  }
471 
472  } else {
473  /* copy dxdotdp over */
474  N_VScale(1.0, dxdotdp.getNVector(ip), sxdot);
475  }
476  J_.multiply(sxdot, sx);
477 }
