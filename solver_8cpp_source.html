69 void Solver::setup(const realtype t0, Model *model, const AmiVector &x0,
70  const AmiVector &dx0, const AmiVectorArray &sx0,
71  const AmiVectorArray &sdx0) const {
72  if (nx() != model->nx_solver || nplist() != model->nplist() ||
73  nquad() != model->nJ * model->nplist()) {
74  resetMutableMemory(model->nx_solver, model->nplist(),
75  model->nJ * model->nplist());
76  }
77  /* Create solver memory object if necessary */
78  allocateSolver();
79  if (!solverMemory)
80  throw AmiException("Failed to allocated solver memory!");
81 
82  /* Initialize CVodes/IDAs solver*/
83  init(t0, x0, dx0);
84 
85  /* Clear diagnosis storage */
86  resetDiagnosis();
87 
88  /* Apply stored tolerances to sundials solver */
89  applyTolerances();
90 
91  /* Set optional inputs */
92  setErrHandlerFn();
93  /* Attaches userdata */
94  setUserData(model);
95  /* specify maximal number of steps */
96  setMaxNumSteps(maxsteps);
97  /* activates stability limit detection */
98  setStabLimDet(stldet);
99 
100  rootInit(model->ne);
101 
102  initializeLinearSolver(model);
103  initializeNonLinearSolver();
104 
105  if (sensi >= SensitivityOrder::first &&
106  sensi_meth > SensitivityMethod::none && model->nx_solver > 0) {
107  auto plist = model->getParameterList();
108  sensInit1(sx0, sdx0);
109  if (sensi_meth == SensitivityMethod::forward && !plist.empty()) {
110  /* Set sensitivity analysis optional inputs */
111  auto par = model->getUnscaledParameters();
112 
113  /* Activate sensitivity calculations and apply tolerances */
114  initializeNonLinearSolverSens(model);
115  setSensParams(par.data(), nullptr, plist.data());
116  applyTolerancesFSA();
117  } else {
118  /* Allocate space for the adjoint computation */
119  adjInit();
120  }
121  }
122 
123  setId(model);
124  setSuppressAlg(true);
125  /* calculate consistent DAE initial conditions (no effect for ODE) */
126  if (model->nt() > 1)
127  calcIC(model->getTimepoint(1));
128 }
