44 void SteadystateProblem::workSteadyStateProblem(Solver *solver, Model *model,
45  int it) {
46 
47  /* process solver handling for pre- or postequilibration */
48  if (it == -1) {
49  /* solver was not run before, set up everything */
50  model->initialize(x_, dx_, sx_, sdx_,
51  solver->getSensitivityOrder() >=
52  SensitivityOrder::first);
53  t_ = model->t0();
54  solver->setup(t_, model, x_, dx_, sx_, sdx_);
55  } else {
56  /* solver was run before, extract current state from solver */
57  solver->writeSolution(&t_, x_, dx_, sx_, xQ_);
58  }
59 
60  /* create a Newton solver obejct */
61  auto newtonSolver = NewtonSolver::getSolver(&t_, &x_, *solver, model);
62 
63  /* Compute steady state and get the computation time */
64  clock_t starttime = clock();
65  findSteadyState(solver, newtonSolver.get(), model, it);
66  cpu_time_ = (double)((clock() - starttime) * 1000) / CLOCKS_PER_SEC;
67 
68  /* Check whether state sensis still need to be computed */
69  if (getSensitivityFlag(model, solver, it, SteadyStateContext::newtonSensi))
70  {
71  try {
72  /* this might still fail, if the Jacobian is singular and
73  simulation did not find a steady state */
74  newtonSolver->computeNewtonSensis(sx_);
75  } catch (NewtonFailure const &) {
76  /* No steady state could be inferred. Store simulation state */
77  storeSimulationState(model, solver->getSensitivityOrder() >=
78  SensitivityOrder::first);
79  throw AmiException("Steady state sensitvitiy computation failed due "
80  "to unsuccessful factorization of RHS Jacobian");
81  }
82  }
83 
84  /* Get output of steady state solver, write it to x0 and reset time
85  if necessary */
86  storeSimulationState(model, getSensitivityFlag(model, solver, it,
87  SteadyStateContext::sensiStorage));
88 }

109 void SteadystateProblem::findSteadyState(Solver *solver,
110  NewtonSolver *newtonSolver,
111  Model *model, int it) {
112  /* First, try to run the Newton solver */
113  steady_state_status_.resize(3, SteadyStateStatus::not_run);
114  findSteadyStateByNewtonsMethod(newtonSolver, model, false);
115 
116  /* Newton solver didn't work, so try to simulate to steady state */
117  if (!checkSteadyStateSuccess())
118  findSteadyStateBySimulation(solver, model, it);
119 
120  /* Simulation didn't work, retry the Newton solver from last sim state. */
121  if (!checkSteadyStateSuccess())
122  findSteadyStateByNewtonsMethod(newtonSolver, model, true);
123 
124  /* Nothing worked, throw an as informative error as possible */
125  if (!checkSteadyStateSuccess())
126  handleSteadyStateFailure(solver, model);
127 }

129 void SteadystateProblem::findSteadyStateByNewtonsMethod(NewtonSolver *newtonSolver,
130  Model *model,
131  bool newton_retry) {
132  int ind = newton_retry ? 2 : 0;
133  try {
134  applyNewtonsMethod(model, newtonSolver, newton_retry);
135  steady_state_status_[ind] = SteadyStateStatus::success;
136  } catch (NewtonFailure const &ex) {
137  /* nothing to be done */
138  switch (ex.error_code) {
139  case AMICI_TOO_MUCH_WORK:
140  steady_state_status_[ind] =
141  SteadyStateStatus::failed_convergence;
142  break;
143  case AMICI_NO_STEADY_STATE:
144  steady_state_status_[ind] =
145  SteadyStateStatus::failed_too_long_simulation;
146  break;
147  case AMICI_SINGULAR_JACOBIAN:
148  steady_state_status_[ind] =
149  SteadyStateStatus::failed_factorization;
150  break;
151  case AMICI_DAMPING_FACTOR_ERROR:
152  steady_state_status_[ind] = SteadyStateStatus::failed_damping;
153  break;
154  default:
155  steady_state_status_[ind] = SteadyStateStatus::failed;
156  break;
157  }
158  }
159 
160  /* copy number of linear steps used */
161  if (max_steps_ > 0) {
162  if (newton_retry) {
163  std::copy_n(newtonSolver->getNumLinSteps().begin(),
164  max_steps_, &numlinsteps_.at(max_steps_));
165  } else {
166  std::copy_n(newtonSolver->getNumLinSteps().begin(),
167  max_steps_, numlinsteps_.begin());
168  }
169  }

172 void SteadystateProblem::findSteadyStateBySimulation(const Solver *solver,
173  Model *model,
174  int it) {
175  /* set starting timepoint for the simulation solver */
176  if (it < 1) /* No previous time point computed, set t = t0 */
177  t_ = model->t0();
178  else /* Carry on simulating from last point */
179  t_ = model->getTimepoint(it - 1);
180 
181  try {
182  if (it < 0) {
183  /* Preequilibration? -> Create a new CVode object for sim */
184  bool integrateSensis = getSensitivityFlag(model, solver, it,
185  SteadyStateContext::solverCreation);
186  auto newtonSimSolver = createSteadystateSimSolver(solver, model,
187  integrateSensis,
188  false);
189  runSteadystateSimulation(newtonSimSolver.get(), model, false);
190  } else {
191  /* Solver was already created, use this one */
192  runSteadystateSimulation(solver, model, false);
193  }
194  steady_state_status_[1] = SteadyStateStatus::success;
195  } catch (NewtonFailure const &ex) {
196  switch (ex.error_code) {
197  case AMICI_TOO_MUCH_WORK:
198  steady_state_status_[1] = SteadyStateStatus::failed_convergence;
199  break;
200  case AMICI_NO_STEADY_STATE:
201  steady_state_status_[1] = SteadyStateStatus::failed_too_long_simulation;
202  break;
203  default:
204  steady_state_status_[1] = SteadyStateStatus::failed;
205  }
206  } catch (AmiException const &) {
207  steady_state_status_[1] = SteadyStateStatus::failed;
208  }

212  Model *model,

214  if (bwd) {
215  /* If preequilibration but not adjoint mode, there's nothing to do */

241 void SteadystateProblem::computeSteadyStateQuadrature(NewtonSolver *newtonSolver,
242  const Solver *solver,
243  Model *model) {
244  /* This routine computes the qudratures:
245  xQB = Integral[ xB(x(t), t, p) * dxdot/dp(x(t), t, p) | dt ]
246  As we're in steady state, we have x(t) = x_ss (x_steadystate), hence
247  xQB = Integral[ xB(x_ss, t, p) | dt ] * dxdot/dp(x_ss, t, p)
248  We therefore compute the integral over xB first and then do a
249  matrix-vector multiplication */
250 
251  /* Try to compute the analytical solution for quadrature algebraically */
252  getQuadratureByLinSolve(newtonSolver, model);
253 
254  /* Analytical solution didn't work, perform simulation instead */
255  if (!hasQuadrature())
256  getQuadratureBySimulation(solver, model);
257 
258  /* If analytic solution and integration did not work, throw an Exception */
259  if (!hasQuadrature())
260  throw AmiException("Steady state backward computation failed: Linear "
261  "system could not be solved (possibly due to singular Jacobian), "
262  "and numerical integration did not equilibrate within maxsteps");
263 }

265 void SteadystateProblem::getQuadratureByLinSolve(NewtonSolver *newtonSolver,
266  Model *model) {
267  /* Computes the integral over the adjoint state xB:
268  If the Jacobian has full rank, this has an anlytical solution, since
269  d/dt[ xB(t) ] = JB^T(x(t), p) xB(t) = JB^T(x_ss, p) xB(t)
270  This linear ODE system with time-constant matrix has the solution
271  xB(t) = exp( t * JB^T(x_ss, p) ) * xB(0)
272  This integral xQ over xB is given as the solution of
273  JB^T(x_ss, p) * xQ = xB(0)
274  So we first try to solve the linear system, if possible. */
275 
276  /* copy content of xB into vector with integral */
277  xQ_.copy(xB_);
278 
279  /* try to solve the linear system */
280  try {
281  /* compute integral over xB and write to xQ */
282  newtonSolver->prepareLinearSystemB(0, -1);
283  newtonSolver->solveLinearSystem(xQ_);
284  /* Compute the quadrature as the inner product xQ * dxotdp */
285  computeQBfromQ(model, xQ_, xQB_);
286  /* set flag that quadratures is available (for processing in rdata) */
287  hasQuadrature_ = true;
288  } catch (NewtonFailure const &) {
289  hasQuadrature_ = false;
290  }
291 }

293 void SteadystateProblem::getQuadratureBySimulation(const Solver *solver,
294  Model *model) {
295  /* If the Jacobian is singular, the integral over xB must be computed
296  by usual integration over time, but simplifications can be applied:
297  x is not time dependent, no forward trajectory is needed. */
298 
299  /* set starting timepoint for the simulation solver */
300  t_ = model->t0();
301  /* xQ was written in getQuadratureByLinSolve() -> reset */
302  xQ_.reset();
303 
304  /* create a new solver object */
305  auto simSolver = createSteadystateSimSolver(solver, model, false, true);
306 
307  /* perform integration and qudrature */
308  try {
309  runSteadystateSimulation(simSolver.get(), model, true);
310  hasQuadrature_ = true;
311  } catch (NewtonFailure const &) {
312  hasQuadrature_ = false;
313  }

359 bool SteadystateProblem::getSensitivityFlag(const Model *model,
360  const Solver *solver,
361  int it, SteadyStateContext context) {
362  /* We need to check whether we need to compute forward sensitivities.
363  Depending on the situation (pre-/postequilibration) and the solver
364  settings, the logic may be involved and is handled here.
365  Most of these boolean operation could be simplified. However,
366  clarity is more important than brevity. */
367 
368  /* Are we running in preequilibration (and hence create)? */
369  bool preequilibration = (it == -1);
370 
371  /* Have we maybe already computed forward sensitivities? */
372  bool forwardSensisAlreadyComputed =
373  solver->getSensitivityOrder() >= SensitivityOrder::first &&
374  steady_state_status_[1] == SteadyStateStatus::success &&
375  model->getSteadyStateSensitivityMode() == SteadyStateSensitivityMode::simulationFSA;
376 
377  /* Do we need forward sensis for postequilibration? */
378  bool needForwardSensisPosteq = !preequilibration &&
379  !forwardSensisAlreadyComputed &&
380  solver->getSensitivityOrder() >= SensitivityOrder::first &&
381  solver->getSensitivityMethod() == SensitivityMethod::forward;
382 
383  /* Do we need forward sensis for preequilibration? */
384  bool needForwardSensisPreeq = preequilibration &&
385  !forwardSensisAlreadyComputed &&
386  solver->getSensitivityMethodPreequilibration() == SensitivityMethod::forward &&
387  solver->getSensitivityOrder() >= SensitivityOrder::first;
388 
389  /* Do we need to do the linear system solve to get forward sensitivities? */
390  bool needForwardSensisNewton =
391  needForwardSensisPreeq || needForwardSensisPosteq;
392 
393  /* When we're creating a new solver object */
394  bool needForwardSensiAtCreation = needForwardSensisPreeq &&
395  model->getSteadyStateSensitivityMode() == SteadyStateSensitivityMode::simulationFSA;
396 
397  /* Check if we need to store sensis */
398  switch (context) {
399  case SteadyStateContext::newtonSensi:
400  return needForwardSensisNewton;
401 
402  case SteadyStateContext::sensiStorage:
403  return needForwardSensisNewton || forwardSensisAlreadyComputed;
404 
405  case SteadyStateContext::solverCreation:
406  return needForwardSensiAtCreation;
407 
408  default:
409  throw AmiException("Requested invalid context in sensitivity "
410  "processing during steady state computation");
411  }
412 }

414 realtype SteadystateProblem::getWrmsNorm(const AmiVector &x,
415  const AmiVector &xdot,
416  realtype atol,
417  realtype rtol,
418  AmiVector &ewt) const {
419  /* Depending on what convergence we want to check (xdot, sxdot, xQBdot)
420  we need to pass ewt[QB], as xdot and xQBdot have different sizes */
421  N_VAbs(x.getNVector(), ewt.getNVector());
422  N_VScale(rtol, ewt.getNVector(), ewt.getNVector());
423  N_VAddConst(ewt.getNVector(), atol, ewt.getNVector());
424  N_VInv(ewt.getNVector(), ewt.getNVector());
425  return N_VWrmsNorm(xdot.getNVector(), ewt.getNVector());
426 }

428 bool SteadystateProblem::checkConvergence(const Solver *solver,
429  Model *model,
430  SensitivityMethod checkSensitivities) {
431  if (checkSensitivities == SensitivityMethod::adjoint) {
432  /* In the adjoint case, only xQB contributes to the gradient, the exact
433  steadystate is less important, as xB = xQdot may even not converge
434  to zero at all. So we need xQBdot, hence compute xQB first. */
435  computeQBfromQ(model, xQ_, xQB_);
436  computeQBfromQ(model, xB_, xQBdot_);
437  wrms_ = getWrmsNorm(xQB_, xQBdot_, solver->getAbsoluteToleranceQuadratures(),
438  solver->getRelativeToleranceQuadratures(), ewtQB_);
439  } else {
440  /* If we're doing a forward simulation (with or without sensitivities:
441  Get RHS and compute weighted error norm */
442  model->fxdot(t_, x_, dx_, xdot_);
443  wrms_ = getWrmsNorm(x_, xdot_, solver->getAbsoluteToleranceSteadyState(),
444  solver->getRelativeToleranceSteadyState(), ewt_);
445  }
446  bool converged = wrms_ < RCONST(1.0);
447 
448  if (checkSensitivities != SensitivityMethod::forward)
449  return converged;
450 
451  /* Forward sensitivities: Compute weighted error norm for their RHS */
452  for (int ip = 0; ip < model->nplist(); ++ip) {
453  if (converged) {
454  sx_ = solver->getStateSensitivity(t_);
455  model->fsxdot(t_, x_, dx_, ip, sx_[ip], dx_, xdot_);
456  wrms_ = getWrmsNorm(
457  x_, xdot_, solver->getAbsoluteToleranceSteadyStateSensi(),
458  solver->getRelativeToleranceSteadyStateSensi(), ewt_);
459  converged = wrms_ < RCONST(1.0);
460  }
461  }
462  return converged;
463 }

476 void SteadystateProblem::applyNewtonsMethod(Model *model,
477  NewtonSolver *newtonSolver,
478  bool newton_retry) {
479  int i_newtonstep = 0;
480  int ix = 0;
481  double gamma = 1.0;
482  bool compNewStep = true;
483 
484  /* initialize output of linear solver for Newton step */
485  delta_.reset();
486 
487  model->fxdot(t_, x_, dx_, xdot_);
488 
489  /* Check for relative error, but make sure not to divide by 0!
490  Ensure positivity of the state */
491  x_newton_ = x_;
492  x_old_ = x_;
493  xdot_old_ = xdot_;
494 
495  wrms_ = getWrmsNorm(x_newton_, xdot_, newtonSolver->atol_,
496  newtonSolver->rtol_, ewt_);
497  bool converged = wrms_ < RCONST(1.0);
498  while (!converged && i_newtonstep < newtonSolver->max_steps) {
499 
500  /* If Newton steps are necessary, compute the inital search direction */
501  if (compNewStep) {
502  try {
503  delta_ = xdot_;
504  newtonSolver->getStep(newton_retry ? 2 : 1, i_newtonstep, delta_);
505  } catch (NewtonFailure const &) {
506  numsteps_.at(newton_retry ? 2 : 0) = i_newtonstep;
507  throw;
508  }
509  }
510 
511  /* Try a full, undamped Newton step */
512  N_VLinearSum(1.0, x_old_.getNVector(), gamma, delta_.getNVector(),
513  x_.getNVector());
514 
515  /* Compute new xdot and residuals */
516  model->fxdot(t_, x_, dx_, xdot_);
517  realtype wrms_tmp = getWrmsNorm(x_newton_, xdot_, newtonSolver->atol_,
518  newtonSolver->rtol_, ewt_);
519 
520  if (wrms_tmp < wrms_) {
521  /* If new residuals are smaller than old ones, update state */
522  wrms_ = wrms_tmp;
523  x_old_ = x_;
524  xdot_old_ = xdot_;
525  /* New linear solve due to new state */
526  compNewStep = true;
527  /* Check residuals vs tolerances */
528  converged = wrms_ < RCONST(1.0);
529 
530  if (converged) {
531  /* Ensure positivity of the found state and recheck if
532  the convergence still holds */
533  bool recheck_convergence = false;
534  for (ix = 0; ix < model->nx_solver; ix++) {
535  if (x_[ix] < 0.0) {
536  x_[ix] = 0.0;
537  recheck_convergence = true;
538  }
539  }
540  if (recheck_convergence) {
541  model->fxdot(t_, x_, dx_, xdot_);
542  wrms_ = getWrmsNorm(x_newton_, xdot_, newtonSolver->atol_,
543  newtonSolver->rtol_, ewt_);
544  converged = wrms_ < RCONST(1.0);
545  }
546  } else if (newtonSolver->damping_factor_mode_==NewtonDampingFactorMode::on) {
547  /* increase dampening factor (superfluous, if converged) */
548  gamma = fmin(1.0, 2.0 * gamma);
549  }
550  } else if (newtonSolver->damping_factor_mode_==NewtonDampingFactorMode::on) {
551  /* Reduce dampening factor and raise an error when becomes too small */
552  gamma = gamma / 4.0;
553  if (gamma < newtonSolver->damping_factor_lower_bound)
554  throw NewtonFailure(AMICI_DAMPING_FACTOR_ERROR,
555  "Newton solver failed: the damping factor "
556  "reached its lower bound");
557 
558  /* No new linear solve, only try new dampening */
559  compNewStep = false;
560  }
561  /* increase step counter */
562  i_newtonstep++;
563  }
564 
565  /* Set return values */
566  numsteps_.at(newton_retry ? 2 : 0) = i_newtonstep;
567  if (!converged)
568  throw NewtonFailure(AMICI_TOO_MUCH_WORK, "applyNewtonsMethod");
569 }

571 void SteadystateProblem::runSteadystateSimulation(const Solver *solver,
572  Model *model,
573  bool backward)
574 {
575  /* Loop over steps and check for convergence
576  NB: This function is used for forward and backward simulation, and may
577  be called by workSteadyStateProblem and workSteadyStateBackwardProblem.
578  Whether we simulate forward or backward in time is reflected by the
579  flag "backward". */
580 
581  /* Do we also have to check for convergence of sensitivities? */
582  SensitivityMethod sensitivityFlag = SensitivityMethod::none;
583  if (solver->getSensitivityOrder() > SensitivityOrder::none &&
584  solver->getSensitivityMethod() > SensitivityMethod::none)
585  sensitivityFlag = SensitivityMethod::forward;
586  /* If flag for forward sensitivity computation by simulation is not set,
587  disable forward sensitivity integration. Sensitivities will be combputed
588  by newonSolver->computeNewtonSensis then */
589  if (model->getSteadyStateSensitivityMode() == SteadyStateSensitivityMode::newtonOnly) {
590  solver->switchForwardSensisOff();
591  sensitivityFlag = SensitivityMethod::none;
592  }
593  if (backward)
594  sensitivityFlag = SensitivityMethod::adjoint;
595 
596  bool converged = checkConvergence(solver, model, sensitivityFlag);
597  int sim_steps = 0;
598 
599  while (!converged) {
600  /* One step of ODE integration
601  reason for tout specification:
602  max with 1 ensures correct direction (any positive value would do)
603  multiplication with 10 ensures nonzero difference and should ensure
604  stable computation value is not important for AMICI_ONE_STEP mode,
605  only direction w.r.t. current t
606  */
607  solver->step(std::max(t_, 1.0) * 10);
608  if (backward) {
609  solver->writeSolution(&t_, xB_, dx_, sx_, xQ_);
610  } else {
611  solver->writeSolution(&t_, x_, dx_, sx_, xQ_);
612  }
613 
614  /* Check for convergence */
615  converged = checkConvergence(solver, model, sensitivityFlag);
616  /* increase counter, check for maxsteps */
617  sim_steps++;
618  if (sim_steps >= solver->getMaxSteps() && !converged) {
619  numsteps_.at(1) = sim_steps;
620  throw NewtonFailure(AMICI_TOO_MUCH_WORK,
621  "exceeded maximum number of steps");
622  }
623  if (t_ >= 1e200 && !converged) {
624  numsteps_.at(1) = sim_steps;
625  throw NewtonFailure(AMICI_NO_STEADY_STATE, "simulated to late time"
626  " point without convergence of RHS");
627  }
628  }
629 
630  /* store information about steps and sensitivities, if necessary */
631  if (backward) {
632  numstepsB_ = sim_steps;
633  } else {
634  numsteps_.at(1) = sim_steps;
635  if (solver->getSensitivityOrder() > SensitivityOrder::none &&
636  model->getSteadyStateSensitivityMode() ==
637  SteadyStateSensitivityMode::simulationFSA)
638  sx_ = solver->getStateSensitivity(t_);
639  }
640 }

642 std::unique_ptr<Solver> SteadystateProblem::createSteadystateSimSolver(
643  const Solver *solver, Model *model, bool forwardSensis, bool backward) const
644 {
645  /* Create new CVode solver object */
646  auto sim_solver = std::unique_ptr<Solver>(solver->clone());
647 
648  switch (solver->getLinearSolver()) {
649  case LinearSolver::dense:
650  break;
651  case LinearSolver::KLU:
652  break;
653  default:
654  throw NewtonFailure(AMICI_NOT_IMPLEMENTED,
655  "invalid solver for steadystate simulation");
656  }
657  /* do we need sensitivities? */
658  if (forwardSensis) {
659  /* need forward to compute sx0 */
660  sim_solver->setSensitivityMethod(SensitivityMethod::forward);
661  } else {
662  sim_solver->setSensitivityMethod(SensitivityMethod::none);
663  sim_solver->setSensitivityOrder(SensitivityOrder::none);
664  }
665  /* use x and sx as dummies for dx and sdx
666  (they wont get touched in a CVodeSolver) */
667  sim_solver->setup(model->t0(), model, x_, x_, sx_, sx_);
668  if (backward) {
669  sim_solver->setup(model->t0(), model, xB_, xB_, sx_, sx_);
670  sim_solver->setupSteadystate(model->t0(), model, x_, x_, xB_, xB_, xQ_);
671  } else {
672  sim_solver->setup(model->t0(), model, x_, x_, sx_, sx_);
673  }
674 
675  return sim_solver;
676 }

678 void SteadystateProblem::computeQBfromQ(Model *model, const AmiVector &yQ,
679  AmiVector &yQB) const {
680  /* Compute the quadrature as the inner product: yQB = dxotdp * yQ */
681 
682  /* reset first, as multiplication add to existing value */
683  yQB.reset();
684  /* multiply */
685  if (model->pythonGenerated) {
686  /* fill dxdotdp with current values */
687  const auto& plist = model->getParameterList();
688  model->fdxdotdp(t_, x_, x_);
689 
690  if (model->ndxdotdp_explicit > 0)
691  model->dxdotdp_explicit.multiply(yQB.getNVector(),
692  yQ.getNVector(), plist, true);
693  if (model->ndxdotdp_implicit > 0)
694  model->dxdotdp_implicit.multiply(yQB.getNVector(),
695  yQ.getNVector(), plist, true);
696  } else {
697  for (int ip=0; ip<model->nplist(); ++ip)
698  yQB[ip] = N_VDotProd(yQ.getNVector(),
699  model->dxdotdp.getNVector(ip));
700  }
701 }
