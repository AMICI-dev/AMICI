243 void Model_DAE::fqBdot(realtype t, N_Vector x, N_Vector dx, N_Vector xB,
244  N_Vector /*dxB*/, N_Vector qBdot) {
245  N_VConst(0.0, qBdot);
246  fdxdotdp(t, x, dx);
247  for (int ip = 0; ip < nplist(); ip++) {
248  for (int ix = 0; ix < nxtrue_solver; ix++)
249  NV_Ith_S(qBdot, ip * nJ) -= NV_Ith_S(xB, ix) * dxdotdp.at(ix, ip);
250  // second order part
251  for (int iJ = 1; iJ < nJ; iJ++)
252  for (int ix = 0; ix < nxtrue_solver; ix++)
253  NV_Ith_S(qBdot, ip * nJ + iJ) -=
254  NV_Ith_S(xB, ix) * dxdotdp.at(ix + iJ * nxtrue_solver, ip) +
255  NV_Ith_S(xB, ix + iJ * nxtrue_solver) * dxdotdp.at(ix, ip);
256  }
257 }

304 void Model_DAE::fsxdot(realtype t, N_Vector x, N_Vector dx, int ip, N_Vector sx,
305  N_Vector sdx, N_Vector sxdot) {
306  if (ip == 0) {
307  // we only need to call this for the first parameter index will be
308  // the same for all remaining
309  fM(t, x);
310  fdxdotdp(t, x, dx);
311  fJSparse(t, 0.0, x, dx, J_.get());
312  }
313 
314  if (pythonGenerated) {
315  // python generated, not yet implemented for DAEs
316  throw AmiException("Wrapping of DAEs is not yet implemented from Python");
317  } else {
318  /* copy dxdotdp over */
319  N_VScale(1.0, dxdotdp.getNVector(ip), sxdot);
320  }
321 
322  J_.multiply(sxdot, sx);
323  N_VScale(-1.0, sdx, sdx);
324  M_.multiply(sxdot, sdx);
325  N_VScale(-1.0, sdx, sdx);
