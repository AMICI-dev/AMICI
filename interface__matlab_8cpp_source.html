506 
507  std::unique_ptr<amici::ExpData> edata;
508  if (nrhs > amici::RHS_DATA && mxGetPr(prhs[amici::RHS_DATA])) {
509  try {

511  } catch (amici::AmiException const& ex) {

515  solver->getSensitivityMethod() == amici::SensitivityMethod::adjoint) {
516  amiciApp.errorF("AMICI:mex:setup","No data provided!");
517  }
