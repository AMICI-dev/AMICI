522  template <class T>
523  void storeJacobianAndDerivativeInReturnData(T const &problem, Model &model)
524  {
525  readSimulationState(problem.getFinalSimulationState(), model);
526 
527  AmiVector xdot(nx_solver);
528  if (!this->xdot.empty() || !this->J.empty())
529  model.fxdot(t_, x_solver_, dx_solver_, xdot);
530 
531  if (!this->xdot.empty())
532  writeSlice(xdot, this->xdot);
533 
534  if (!this->J.empty()) {
535  SUNMatrixWrapper J(nx_solver, nx_solver);
536  model.fJ(t_, 0.0, x_solver_, dx_solver_, xdot, J.get());
537  // CVODES uses colmajor, so we need to transform to rowmajor
538  for (int ix = 0; ix < model.nx_solver; ix++)
539  for (int jx = 0; jx < model.nx_solver; jx++)
540  this->J.at(ix * model.nx_solver + jx) =
541  J.data()[ix + model.nx_solver * jx];
542  }
543  }
