523  template <class T>
524  void storeJacobianAndDerivativeInReturnData(T const &problem, Model &model)
525  {
526  readSimulationState(problem.getFinalSimulationState(), model);
527 
528  AmiVector xdot(nx_solver);
529  if (!this->xdot.empty() || !this->J.empty())
530  model.fxdot(t_, x_solver_, dx_solver_, xdot);
531 
532  if (!this->xdot.empty())
533  writeSlice(xdot, this->xdot);
534 
535  if (!this->J.empty()) {
536  SUNMatrixWrapper J(nx_solver, nx_solver);
537  model.fJ(t_, 0.0, x_solver_, dx_solver_, xdot, J.get());
538  // CVODES uses colmajor, so we need to transform to rowmajor
539  for (int ix = 0; ix < model.nx_solver; ix++)
540  for (int jx = 0; jx < model.nx_solver; jx++)
541  this->J.at(ix * model.nx_solver + jx) =
542  J.data()[ix + model.nx_solver * jx];
543  }
544  }
